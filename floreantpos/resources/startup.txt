### get the source

git clone https://github.com/fijiaaron/FloreantPOS.git
git checkout scipio

### build the project

mvn clean install

### copy needed files 

cd target 
set TARGET=floreantpos-1.2-SNAPSHOT-bin\floreantpos-1.2-SNAPSHOT

cp classes\scipio.properties %TARGET%
cp classes\floreantpos.properties %TARGET%
cp classes\update_posdb.bat %TARGET%
cp -r classes\database %TARGET%

### you can zip up to create a distribution package from target at this point
### but to run, you still need to update the database

set RUN=\temp\scipio
cp -r %TARGET% %RUN%
cd %RUN%\floreantpos-1.2-SNAPSHOT


### unzip and start the database 

cd sample-db
unzip posdb.zip
start-server.bat

### run the database update script

cd %RUN%\floreantpos-1.2-SNAPSHOT
update_posdb.bat

### if the floreantpos jar has been renamed, edit floreantpos.bat to match
### then run floreantpos

floreantpos.bat
